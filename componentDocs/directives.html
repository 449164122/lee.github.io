# 防抖

v-debounce

v-debounce="方法名"
// 如
v-debounce="handleConfirm"

// 默认防抖时间为1000毫秒，可修改
v-debounce:300="handleConfirm"

# 节流

v-throttle

v-throttle="方法名"
// 如
v-throttle="handleConfirm"

// 默认节流时间为1000毫秒，可修改
v-throttle:300="handleConfirm"
